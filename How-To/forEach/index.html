How To...

/*
  ..use a 'forEach' loop
*/


// Syntax
array.forEach(element => {
  console.log(element)
});


// Example
let buttons = document.querySelectorAll('.button');
buttons.forEach(button => {
  button.addEventListener('click', () => {
    console.log("Button Clicked");
  });
});

// OUTPUT: 'Button Clicked'
